cmake .. -DLUAU_EXTERN_C=ON \
         -DLUAU_BUILD_SHARED=OFF \
         -DCMAKE_POSITION_INDEPENDENT_CODE=ON \
         -DCMAKE_BUILD_TYPE=Release


cmake --build . --config Release --target Luau.VM Luau.Compiler Luau.Ast Luau.Bytecode Luau.Common

g++ -shared -o libluau.so \
    -Wl,--whole-archive \
    libLuau.VM.a \
    libLuau.Compiler.a \
    libLuau.Ast.a \
    libLuau.Bytecode.a \c
    libLuau.Common.a \
    -Wl,--no-whole-archiv
